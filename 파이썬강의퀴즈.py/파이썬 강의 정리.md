class Unit:
    def __init__(self, name, hp):
        self.name = name
        self.hp = hp
class AttackUnit(Unit):
    def __init__(self, name, hp, damage):
        Unit.__init__(self, name, hp)
        self.damage = damage
        #유닛(부모클래스)으로부터 상속받음 
	def attack(self, location):
